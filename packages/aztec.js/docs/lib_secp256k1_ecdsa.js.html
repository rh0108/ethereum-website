'use strict';/**
 * Create and validate ECDSA signatures, uses elliptic.js library
 *
 * @namespace ecdsa
 * @memberof module:secp256k1
 */var BN=require('bn.js'),web3Utils=require('web3-utils'),Web3EthAccounts=require('web3-eth-accounts'),secp256k1=require('./index'),web3EthAccounts=new Web3EthAccounts,ecdsa={};/**
 * Convert an Ethereum public key into an address
 *
 * @method accountFromPublicKey
 * @memberof module:secp256k1.ecdsa
 * @param {string} publicKey hex-string formatted public key (uncompressed)
 * @returns {string} address
 *//**
 * Sign a message hash with a given private key
 *
 * @method signMessage
 * @memberof module:secp256k1.ecdsa
 * @param {string} hash hex-string formatted message hash
 * @param {string} privateKey hex-string formatted private key
 * @returns {string[]} ECDSA signature parameters [v, r, s], formatted as 32-byte wide hex-strings
 *//**
 * Verify an ECDSA signature against a publickey
 *
 * @method verifyMessage
 * @memberof module:secp256k1.ecdsa
 * @param {string} hash hex-string formatted message hash
 * @param {string} r hex-string formatted ECDSA parameter r
 * @param {string} s hex-string formatted ECDSA parameter s
 * @param {string} publicKey hex-string formatted public key (uncompressed)
 * @returns {bool} true if signature signed by publicKey, false otherwise
 *//**
 * Recover the signing key of an ECDSA signature
 *
 * @method recoverPublicKey
 * @memberof module:secp256k1.ecdsa
 * @param {string} hash hex-string formatted message hash
 * @param {string} r hex-string formatted ECDSA parameter r
 * @param {string} s hex-string formatted ECDSA parameter s
 * @param {string} vn hex-string ECDSA parameter v
 * @returns {Object} elliptic.js public key object of message signer
 *//**
 * Compares signatures from this module with those signed by web3. For debug and test purposes  
 * (we don't use web3 because we want a different preamble for eip712 signatures)
 *
 * @method web3Comparison
 * @memberof module:secp256k1.ecdsa
 * @returns {Object} ecdsa module signature and web3 signature
 */ecdsa.accountFromPublicKey=function(a){var b;if('string'==typeof a)b=web3Utils.sha3('0x'+a.slice(4));else{var c=secp256k1.ec.keyFromPublic(a),d='0x'+c.getPublic(!1,'hex').slice(2);b=web3Utils.sha3(d)}var e=web3Utils.toChecksumAddress('0x'+b.slice(-40));return e},ecdsa.signMessage=function(a,b){var c=secp256k1.ec.keyFromPrivate(Buffer.from(b.slice(2),'hex')).sign(Buffer.from(a.slice(2),'hex'),{canonical:!0});return['0x'+web3Utils.padLeft((+(27+ +c.recoveryParam)).toString(16),64),'0x'+web3Utils.padLeft(c.r.toString(16),64),'0x'+web3Utils.padLeft(c.s.toString(16),64)]},ecdsa.verifyMessage=function(a,b,c,d){var e=new BN(b.slice(2),16),f=new BN(c.slice(2),16);return secp256k1.ec.verify(a.slice(2),{r:e,s:f},secp256k1.ec.keyFromPublic(d.slice(2),'hex'))},ecdsa.recoverPublicKey=function(a,b,c,d){var e=new BN(b.slice(2),16),f=new BN(c.slice(2),16),g=new BN(d.slice(2),16).toNumber(),h=secp256k1.ec.recoverPubKey(Buffer.from(web3Utils.padLeft(a.slice(2),64),'hex'),{r:e,s:f},2>g?g:1-g%2);return h},ecdsa.web3Comparison=function(){var a=web3EthAccounts.create(),b=a.privateKey,c=a.address,d=a.sign(a.address,''),e=Buffer.from(web3Utils.hexToBytes(c,'hex')),f=Buffer.from('\x19Ethereum Signed Message:\n'+e.length),g=Buffer.concat([f,e]),h=web3Utils.sha3(g),i=ecdsa.signMessage(h,b);return{result:i,web3Sig:d}},module.exports=ecdsa;
